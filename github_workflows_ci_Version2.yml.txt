name: Install Hugging Face Model

on:
  push:
    branches: [main, master]
  pull_request:
    branches: [main, master]

jobs:
  install-model:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python 3.10
        uses: actions/setup-python@v5
        with:
          python-version: '3.10'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install huggingface_hub git-lfs

      - name: Install git-lfs
        run: |
          git lfs install

      - name: Clone Hugging Face model repo and download files
        run: |
          git clone https://huggingface.co/Salesforce/codet5-small model/
          huggingface-cli download Salesforce/codet5-small --local-dir model/
          ls model/

      - name: List contents of downloaded model directory
        run: |
          ls -l model/

      # Optional: Run your Python script using the model
      # - name: Run script using model
      #   run: |
      #     python your_script.py

```
