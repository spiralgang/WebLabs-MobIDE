name: Third-party dependency scan (Renovate dry-run)

on:
  workflow_dispatch: {}
  schedule:
    - cron: '0 4 * * 0' # weekly Sunday 04:00 UTC

permissions:
  contents: read
  pull-requests: write

jobs:
  renovate-scan:
    runs-on: ubuntu-latest
    timeout-minutes: 45
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install renovate CLI
        run: |
          set -e
          npm install -g renovate@latest || true

      - name: Run Renovate in dry-run mode and save JSON output
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
        run: |
          set -e || true
          renovate --config-file .github/renovate.json --dry-run --log-level debug --autodiscover > renovate-output.json 2>&1 || true
          if [ ! -f renovate-output.json ]; then echo "{\"renovate\":\"no-output\"}" > renovate-output.json; fi

      - name: Upload renovate report
        uses: actions/upload-artifact@v4
        with:
          name: renovate-report
          path: renovate-output.json

      - name: Summary: write brief counts to job summary
        shell: bash
        run: |
          echo "### Renovate dry-run summary" >> $GITHUB_STEP_SUMMARY || true
          CNT=$(grep -c "INFO" renovate-output.json || true)
          echo "- Approximate log INFO lines: ${CNT}" >> $GITHUB_STEP_SUMMARY || true
