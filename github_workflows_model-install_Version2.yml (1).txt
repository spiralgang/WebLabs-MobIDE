name: Install Hugging Face Model (Audit-Proof)

on:
  push:
    branches: [main, master]
  pull_request:
    branches: [main, master]

jobs:
  fetch-model:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python 3.10
        uses: actions/setup-python@v5
        with:
          python-version: '3.10'

      - name: Install pip dependencies
        run: |
          python -m pip install --upgrade pip
          pip install huggingface_hub

      - name: Install git-lfs
        run: |
          sudo apt-get update
          sudo apt-get install -y git-lfs
          git lfs install

      - name: Clone Hugging Face model repo
        run: |
          git clone https://huggingface.co/Salesforce/codet5-small model/

      - name: Download model files with huggingface-cli (redundancy for audit)
        run: |
          huggingface-cli download Salesforce/codet5-small --local-dir model/

      - name: List contents of model directory
        run: |
          ls -lh model/
